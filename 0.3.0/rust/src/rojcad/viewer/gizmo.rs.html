29fn axis_color(index: usize) -> [f32; 4] {
30    match index {
31        0 => [0.941, 0.216, 0.322, 1.0],
32        1 => [0.463, 0.702, 0.086, 1.0],
33        2 => [0.176, 0.537, 0.941, 1.0],
34        _ => [0.5, 0.5, 0.5, 1.0],
35    }

38fn to_array(v: Vec3) -> [f32; 3] {
39    [v.x, v.y, v.z]

42fn generate_cylinder(
43    center: Vec3,
44    tip: Vec3,
45    color: [f32; 4],
46    radius: f32,
47    segments: u32,
48) -> Vec<GizmoVertex> {
49    let dir = (tip - center).normalize();
50    let up = if dir.y.abs() < 0.9 { Vec3::Y } else { Vec3::X };
51    let right = dir.cross(up).normalize();
52    let forward = right.cross(dir);
53
54    let mut verts = Vec::with_capacity((segments * 6) as usize);
55
56    for i in 0..segments {
57        let angle0 = (i as f32 / segments as f32) * std::f32::consts::TAU;
58        let angle1 = ((i + 1) as f32 / segments as f32) * std::f32::consts::TAU;
59
60        let (sin0, cos0) = angle0.sin_cos();
61        let (sin1, cos1) = angle1.sin_cos();
62
63        let r0 = right * cos0 + forward * sin0;
64        let r1 = right * cos1 + forward * sin1;
65
66        let c0 = center + r0 * radius;
67        let c1 = center + r1 * radius;
68        let t0 = tip + r0 * radius;
69        let t1 = tip + r1 * radius;
70
71        verts.push(GizmoVertex {
72            position: to_array(c0),
73            color,
74        });
75        verts.push(GizmoVertex {
76            position: to_array(t0),
77            color,
78        });
79        verts.push(GizmoVertex {
80            position: to_array(c1),
81            color,
82        });
83        verts.push(GizmoVertex {
84            position: to_array(t0),
85            color,
86        });
87        verts.push(GizmoVertex {
88            position: to_array(t1),
89            color,
90        });
91        verts.push(GizmoVertex {
92            position: to_array(c1),
93            color,
94        });
95    }
96
97    verts
98}

100fn generate_sphere(center: Vec3, color: [f32; 4], radius: f32) -> Vec<GizmoVertex> {
101    let mut verts = Vec::new();
102
103    for lat in 0..SPHERE_LAT {
104        let theta0 = (lat as f32 / SPHERE_LAT as f32) * std::f32::consts::PI;
105        let theta1 = ((lat + 1) as f32 / SPHERE_LAT as f32) * std::f32::consts::PI;
106        let sin_t0 = theta0.sin();
107        let cos_t0 = theta0.cos();
108        let sin_t1 = theta1.sin();
109        let cos_t1 = theta1.cos();
110
111        for lon in 0..SPHERE_LON {
112            let phi0 = (lon as f32 / SPHERE_LON as f32) * std::f32::consts::TAU;
113            let phi1 = ((lon + 1) as f32 / SPHERE_LON as f32) * std::f32::consts::TAU;
114            let sin_p0 = phi0.sin();
115            let cos_p0 = phi0.cos();
116            let sin_p1 = phi1.sin();
117            let cos_p1 = phi1.cos();
118
119            let v0 = Vec3::new(sin_t0 * cos_p0, cos_t0, sin_t0 * sin_p0) * radius + center;
120            let v1 = Vec3::new(sin_t0 * cos_p1, cos_t0, sin_t0 * sin_p1) * radius + center;
121            let v2 = Vec3::new(sin_t1 * cos_p0, cos_t1, sin_t1 * sin_p0) * radius + center;
122            let v3 = Vec3::new(sin_t1 * cos_p1, cos_t1, sin_t1 * sin_p1) * radius + center;
123
124            verts.push(GizmoVertex {
125                position: to_array(v0),
126                color,
127            });
128            verts.push(GizmoVertex {
129                position: to_array(v2),
130                color,
131            });
132            verts.push(GizmoVertex {
133                position: to_array(v1),
134                color,
135            });
136            verts.push(GizmoVertex {
137                position: to_array(v1),
138                color,
139            });
140            verts.push(GizmoVertex {
141                position: to_array(v2),
142                color,
143            });
144            verts.push(GizmoVertex {
145                position: to_array(v3),
146                color,
147            });
148        }
149    }
150
151    verts
152}

154fn build_letter_mesh_3d(
155    center: Vec3,
156    right: Vec3,
157    up: Vec3,
158    letter: u8,
159    depth_bias: Vec3,
160) -> Vec<GizmoVertex> {
161    let hw = 0.09;
162    let hh = 0.12;
163    let lw = 0.03;
164    let color = [0.15, 0.15, 0.15, 1.0];
165
166    let pos = center + depth_bias;
167
168    fn bar(center: Vec3, r: Vec3, u: Vec3, from: Vec3, to: Vec3, w: f32) -> Vec<Vec3> {
169        let dir = (to - from).normalize();
170        let perp = Vec3::new(-dir.y, dir.x, 0.0) * (w * 0.5);
171        let a0 = from - perp;
172        let a1 = from + perp;
173        let b0 = to - perp;
174        let b1 = to + perp;
175        let to_w = |p: Vec3| center + r * p.x + u * p.y;
176        vec![to_w(a0), to_w(b0), to_w(a1), to_w(b0), to_w(b1), to_w(a1)]
177    }
178
179    let bars: Vec<[Vec3; 2]> = match letter {
180        b'X' => vec![
181            [Vec3::new(-hw, -hh, 0.0), Vec3::new(hw, hh, 0.0)],
182            [Vec3::new(hw, -hh, 0.0), Vec3::new(-hw, hh, 0.0)],
183        ],
184        b'Y' => vec![
185            [Vec3::new(-hw, hh, 0.0), Vec3::new(0.0, 0.0, 0.0)],
186            [Vec3::new(hw, hh, 0.0), Vec3::new(0.0, 0.0, 0.0)],
187            [Vec3::new(0.0, 0.0, 0.0), Vec3::new(0.0, -hh, 0.0)],
188        ],
189        b'Z' => vec![
190            [Vec3::new(-hw, hh, 0.0), Vec3::new(hw, hh, 0.0)],
191            [Vec3::new(hw, hh, 0.0), Vec3::new(-hw, -hh, 0.0)],
192            [Vec3::new(-hw, -hh, 0.0), Vec3::new(hw, -hh, 0.0)],
193        ],
194        _ => vec![],
195    };
196
197    let mut verts = Vec::new();
198    for [from, to] in bars {
199        let pts = bar(pos, right, up, from, to, lw);
200        for p in pts {
201            verts.push(GizmoVertex {
202                position: to_array(p),
203                color,
204            });
205        }
206    }
207    verts
208}

350    fn create_depth(device: &wgpu::Device, size: u32) -> (wgpu::Texture, wgpu::TextureView) {
351        let size = size.max(1);
352        let texture = device.create_texture(&wgpu::TextureDescriptor {
353            label: Some("gizmo_depth"),
354            size: wgpu::Extent3d {
355                width: size,
356                height: size,
357                depth_or_array_layers: 1,
358            },
359            mip_level_count: 1,
360            sample_count: 1,
361            dimension: wgpu::TextureDimension::D2,
362            format: wgpu::TextureFormat::Depth32Float,
363            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
364            view_formats: &[],
365        });
366        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
367        (texture, view)
368    }

370    fn build_static_vertices(hovered: Option<usize>) -> Vec<GizmoVertex> {
371        let axis_dirs: [Vec3; 3] = [Vec3::X, Vec3::Y, Vec3::Z];
372        let mut verts = Vec::new();
373
374        for (i, &dir) in axis_dirs.iter().enumerate() {
375            let tip = dir * AXIS_LEN;
376            let color = axis_color(i);
377
378            verts.extend(generate_cylinder(
379                Vec3::ZERO,
380                tip,
381                color,
382                LINE_WIDTH * 0.5,
383                SPHERE_LON,
384            ));
385
386            let r = if hovered == Some(i) {
387                CIRCLE_RADIUS * HOVER_SCALE
388            } else {
389                CIRCLE_RADIUS
390            };
391            verts.extend(generate_sphere(tip, color, r));
392        }
393
394        verts.extend(generate_sphere(
395            Vec3::ZERO,
396            [0.4, 0.4, 0.4, 1.0],
397            LINE_WIDTH,
398        ));
399
400        verts
401    }

410    pub fn update_uniforms(
411        &mut self,
412        queue: &wgpu::Queue,
413        device: &wgpu::Device,
414        main_camera: &OrbitCamera,
415    ) {
416        let forward = main_camera.forward();
417        let eye = forward * 5.0;
418        let gizmo_view = DMat4::look_at_lh(eye, DVec3::ZERO, DVec3::Y);
419        let gizmo_proj = DMat4::orthographic_lh(
420            -GIZMO_VIEWPORT_SIZE,
421            GIZMO_VIEWPORT_SIZE,
422            -GIZMO_VIEWPORT_SIZE,
423            GIZMO_VIEWPORT_SIZE,
424            -10.0,
425            10.0,
426        );
427        let view_proj = gizmo_proj * gizmo_view;
428
429        let uniforms = GizmoUniforms {
430            view_proj: view_proj.to_cols_array_2d().map(|r| r.map(|v| v as f32)),
431        };
432        queue.write_buffer(&self.uniform_buffer, 0, bytemuck::bytes_of(&uniforms));
433
434        let f = (DVec3::ZERO - eye).normalize();
435        let right = f.cross(DVec3::Y).normalize();
436        let up = right.cross(f).normalize();
437
438        let right_v3 = Vec3::new(right.x as f32, right.y as f32, right.z as f32);
439        let up_v3 = Vec3::new(up.x as f32, up.y as f32, up.z as f32);
440        let forward_v3 = Vec3::new(f.x as f32, f.y as f32, f.z as f32);
441        let depth_bias = -forward_v3 * LETTER_DEPTH_BIAS;
442
443        let axis_dirs = [Vec3::X, Vec3::Y, Vec3::Z];
444        let letters = [b'X', b'Y', b'Z'];
445
446        let mut letter_verts = Vec::new();
447        for i in 0..3 {
448            let center = axis_dirs[i] * AXIS_LEN;
449            let r = if i == 2 { -right_v3 } else { right_v3 };
450            letter_verts.extend(build_letter_mesh_3d(
451                center, r, up_v3, letters[i], depth_bias,
452            ));
453        }
454
455        let all_verts: Vec<GizmoVertex> = Self::build_static_vertices(self.hovered)
456            .into_iter()
457            .chain(letter_verts)
458            .collect();
459
460        self.vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
461            label: Some("gizmo_vertices"),
462            contents: bytemuck::cast_slice(&all_verts),
463            usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
464        });
465        self.num_vertices = all_verts.len() as u32;
466    }

469        if self.hovered == hovered {
470            return;
471        }
472        self.hovered = hovered;

475    pub fn render(&self, pass: &mut wgpu::RenderPass) {
476        pass.set_pipeline(&self.pipeline);
477        pass.set_bind_group(0, &self.bind_group, &[]);
478        pass.set_vertex_buffer(0, self.vertex_buffer.slice(..));
479        pass.draw(0..self.num_vertices, 0..1);
480    }
