61const ANIMATION_DURATION: f64 = 0.5;
62
63struct CameraAnimation {
64    active: bool,
65    start_yaw: f64,
66    start_pitch: f64,
67    target_yaw: f64,
68    target_pitch: f64,
69    elapsed: f64,
70}
71
72fn ease_in_out(t: f64) -> f64 {
73    t * t * (3.0 - 2.0 * t)
74}
75
76impl CameraAnimation {
77    fn new() -> Self {
78        Self {
79            active: false,
80            start_yaw: 0.0,
81            start_pitch: 0.0,
82            target_yaw: 0.0,
83            target_pitch: 0.0,
84            elapsed: 0.0,
85        }
86    }
87
88    fn start(&mut self, camera: &OrbitCamera, target_yaw: f64, target_pitch: f64) {
89        self.active = true;
90        self.start_yaw = camera.yaw;
91        self.start_pitch = camera.pitch;
92        self.target_yaw = target_yaw;
93        self.target_pitch = target_pitch;
94        self.elapsed = 0.0;
95    }
96
97    fn update(&mut self, camera: &mut OrbitCamera, dt: f64) {
98        if !self.active {
99            return;
100        }
101        self.elapsed += dt;
102        let t = (self.elapsed / ANIMATION_DURATION).clamp(0.0, 1.0);
103        let e = ease_in_out(t);
104        camera.yaw = self.start_yaw + (self.target_yaw - self.start_yaw) * e;
105        camera.pitch = self.start_pitch + (self.target_pitch - self.start_pitch) * e;
106        if t >= 1.0 {
107            camera.yaw = self.target_yaw;
108            camera.pitch = self.target_pitch;
109            self.active = false;
110            camera.perspective = false;
111        }
112    }
113
114    fn stop(&mut self) {
115        self.active = false;
116    }
117}

121/// Smoothly animates camera target, radius, yaw, and pitch for fit-to-shape.
122struct FitAnimation {
123    start_target: DVec3,
124    end_target: DVec3,
125    start_radius: f64,
126    end_radius: f64,
127    start_yaw: f64,
128    end_yaw: f64,
129    start_pitch: f64,
130    end_pitch: f64,
131    elapsed: f64,
132    duration: f64,
133}
134
135impl FitAnimation {
136    fn new(
137        start_target: DVec3,
138        end_target: DVec3,
139        start_radius: f64,
140        end_radius: f64,
141        start_yaw: f64,
142        end_yaw: f64,
143        start_pitch: f64,
144        end_pitch: f64,
145    ) -> Self {
146        Self {
147            start_target,
148            end_target,
149            start_radius,
150            end_radius,
151            start_yaw,
152            end_yaw,
153            start_pitch,
154            end_pitch,
155            elapsed: 0.0,
156            duration: ANIMATION_DURATION,
157        }
158    }
159
160    fn update(&mut self, camera: &mut OrbitCamera, dt: f64) -> bool {
161        self.elapsed += dt;
162        let t = (self.elapsed / self.duration).clamp(0.0, 1.0);
163        let e = ease_in_out(t);
164
165        camera.target = self.start_target.lerp(self.end_target, e);
166        camera.radius = self.start_radius + (self.end_radius - self.start_radius) * e;
167        camera.yaw = self.start_yaw + (self.end_yaw - self.start_yaw) * e;
168        camera.pitch = self.start_pitch + (self.end_pitch - self.start_pitch) * e;
169
170        t >= 1.0
171    }
172}

176/// GPU-side representation of a single shape's mesh.
177pub struct CadMesh {
178    pub vertex_buffer: wgpu::Buffer,
179    pub index_buffer: wgpu::Buffer,
180    pub num_indices: u32,
181    pub shape_id: ShapeId,
182    #[allow(dead_code)]
183    pub color_buffer: wgpu::Buffer,
184    pub color_bind_group: wgpu::BindGroup,
185}
186
187impl CadMesh {
188    pub fn new(
189        device: &wgpu::Device,
190        mesh: &MeshData,
191        shape_id: ShapeId,
192        color: Option<[f64; 3]>,
193        color_bind_group_layout: &wgpu::BindGroupLayout,
194    ) -> Self {
195        debug_assert_eq!(
196            mesh.normals.len(),
197            mesh.vertices.len(),
198            "CadMesh::new: normals count ({}) must match vertices count ({})",
199            mesh.normals.len(),
200            mesh.vertices.len()
201        );
202        let interleaved: Vec<MeshVertex> = mesh
203            .vertices
204            .iter()
205            .zip(mesh.normals.iter())
206            .map(|(pos, norm)| MeshVertex {
207                position: *pos,
208                normal: *norm,
209            })
210            .collect();
211        let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
212            label: Some(&format!("cad_mesh_{}_vertex", shape_id)),
213            contents: bytemuck::cast_slice(&interleaved),
214            usage: wgpu::BufferUsages::VERTEX,
215        });
216
217        let index_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
218            label: Some(&format!("cad_mesh_{}_index", shape_id)),
219            contents: bytemuck::cast_slice(&mesh.indices),
220            usage: wgpu::BufferUsages::INDEX,
221        });
222
223        // Per-mesh color uniform
224        let c = color.unwrap_or([0.75, 0.75, 0.75]);
225        let color_uniform: [f32; 4] = [c[0] as f32, c[1] as f32, c[2] as f32, 1.0];
226        let color_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
227            label: Some(&format!("cad_mesh_{}_color", shape_id)),
228            contents: bytemuck::bytes_of(&color_uniform),
229            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
230        });
231
232        let color_bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
233            label: Some(&format!("cad_mesh_{}_color_bg", shape_id)),
234            layout: color_bind_group_layout,
235            entries: &[wgpu::BindGroupEntry {
236                binding: 0,
237                resource: color_buffer.as_entire_binding(),
238            }],
239        });
240
241        CadMesh {
242            vertex_buffer,
243            index_buffer,
244            num_indices: mesh.indices.len() as u32,
245            shape_id,
246            color_buffer,
247            color_bind_group,
248        }
249    }
250}

455    pub fn render(
456        &self,
457        pass: &mut wgpu::RenderPass,
458        selected_ids: &HashSet<ShapeId>,
459        highlighted_id: Option<ShapeId>,
460    ) {
461        pass.set_bind_group(0, &self.uniform_bind_group, &[]);
462        for mesh in &self.meshes {
463            let use_highlight = selected_ids.contains(&mesh.shape_id)
464                || highlighted_id.is_some_and(|hid| hid == mesh.shape_id);
465            if use_highlight {
466                pass.set_pipeline(&self.highlight_pipeline);
467            } else {
468                pass.set_pipeline(&self.render_pipeline);
469            }
470            pass.set_bind_group(1, &mesh.color_bind_group, &[]);
471            pass.set_vertex_buffer(0, mesh.vertex_buffer.slice(..));
472            pass.set_index_buffer(mesh.index_buffer.slice(..), wgpu::IndexFormat::Uint32);
473            pass.draw_indexed(0..mesh.num_indices, 0, 0..1);
474        }
475    }

644    fn build_pipeline(
645        device: &wgpu::Device,
646        layout: &wgpu::PipelineLayout,
647        surface_format: wgpu::TextureFormat,
648        depth_format: wgpu::TextureFormat,
649        vertex_entry: &str,
650        fragment_entry: &str,
651        topology: wgpu::PrimitiveTopology,
652        vertex_buffers: &[wgpu::VertexBufferLayout<'_>],
653        depth_compare: wgpu::CompareFunction,
654        depth_bias: wgpu::DepthBiasState,
655    ) -> wgpu::RenderPipeline {
656        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
657            label: Some("edge shader"),
658            source: wgpu::ShaderSource::Wgsl(std::borrow::Cow::Borrowed(include_str!("edge.wgsl"))),
659        });
660
661        device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
662            label: Some(fragment_entry),
663            layout: Some(layout),
664            vertex: wgpu::VertexState {
665                module: &shader,
666                entry_point: Some(vertex_entry),
667                buffers: vertex_buffers,
668                compilation_options: wgpu::PipelineCompilationOptions::default(),
669            },

726    pub fn render(
727        &self,
728        pass: &mut wgpu::RenderPass,
729        inactive_buffer: &wgpu::Buffer,
730        inactive_num_instances: u32,
731        active_buffer: &wgpu::Buffer,
732        active_num_instances: u32,
733        show_dashed: bool,
734    ) {
735        pass.set_bind_group(0, &self.uniform_bind_group, &[]);
736
737        let show_inactive = SHOW_INACTIVE_EDGES.load(std::sync::atomic::Ordering::Relaxed);
738        let show_active = SHOW_ACTIVE_EDGES.load(std::sync::atomic::Ordering::Relaxed);
739
740        // Inactive edges (light grey) — 4 verts per instance (TriangleStrip quad)
741        if show_inactive && inactive_num_instances > 0 {
742            pass.set_vertex_buffer(0, inactive_buffer.slice(..));
743            if show_dashed {
744                pass.set_pipeline(&self.inactive_dashed_pipeline);
745                pass.draw(0..4, 0..inactive_num_instances);
746            }
747            pass.set_pipeline(&self.inactive_solid_pipeline);
748            pass.draw(0..4, 0..inactive_num_instances);
749        }
750
751        // Active edges (light blue, rendered on top)
752        if show_active && active_num_instances > 0 {
753            pass.set_vertex_buffer(0, active_buffer.slice(..));
754            if show_dashed {
755                pass.set_pipeline(&self.active_dashed_pipeline);
756                pass.draw(0..4, 0..active_num_instances);
757            }
758            pass.set_pipeline(&self.active_solid_pipeline);
759            pass.draw(0..4, 0..active_num_instances);
760        }
761    }

766/// Flatten a slice of segment endpoint pairs into a vertex buffer.
767fn segments_to_vertices(segments: &[[[f32; 3]; 2]]) -> Vec<[f32; 3]> {
768    let mut verts = Vec::with_capacity(segments.len() * 2);
769    for seg in segments {
770        verts.push(seg[0]);
771        verts.push(seg[1]);
772    }
773    verts
774}

805fn generate_grid_segments() -> Vec<[[f32; 3]; 2]> {
806    let mut segments = Vec::new();
807    let half_size = 100.0;
808    let major_step = 10.0;
809
810    // Central axes (X and Z through origin)
811    segments.push([[-half_size, 0.0, 0.0], [half_size, 0.0, 0.0]]);
812    segments.push([[0.0, 0.0, -half_size], [0.0, 0.0, half_size]]);
813
814    // Major and minor grid lines on XZ plane
815    let mut z = major_step;
816    while z <= half_size {
817        // Lines parallel to X axis
818        segments.push([[-half_size, 0.0, z], [half_size, 0.0, z]]);
819        segments.push([[-half_size, 0.0, -z], [half_size, 0.0, -z]]);
820        // Lines parallel to Z axis
821        segments.push([[z, 0.0, -half_size], [z, 0.0, half_size]]);
822        segments.push([[-z, 0.0, -half_size], [-z, 0.0, half_size]]);
823        z += major_step;
824    }
825
826    segments
827}

1296    fn handle_click(state: &mut ViewerState, viewer_tx: &Sender<ViewerToRepl>) {
1297        // Shape picking from mouse click
1298        let aspect = state.size.width as f64 / state.size.height.max(1) as f64;
1299        let view_proj = state.camera.matrix(aspect);
1300        let inv_view_proj = view_proj.inverse();
1301
1302        // Normalize mouse position to NDC [-1, 1]
1303        let ndc_x = (state.mouse_pos.x / state.size.width as f64) * 2.0 - 1.0;
1304        let ndc_y = 1.0 - (state.mouse_pos.y / state.size.height as f64) * 2.0;
1305
1306        let near_h = inv_view_proj * glam::DVec4::new(ndc_x, ndc_y, -1.0, 1.0);
1307        let far_h = inv_view_proj * glam::DVec4::new(ndc_x, ndc_y, 1.0, 1.0);
1308        let near = near_h.truncate() / near_h.w;
1309        let far = far_h.truncate() / far_h.w;
1310
1311        let origin = near;
1312        let dir = (far - near).normalize();
1313
1314        // Collect visible mesh data for picking
1315        let registry = global_shape_registry();
1316        let visible = registry.visible_shapes();
1317        let mesh_refs: Vec<(u64, &MeshData)> = visible
1318            .iter()
1319            .filter_map(|e| e.mesh.as_ref().map(|m| (e.shape_id, m)))
1320            .collect();
1321
1322        let ctrl = state.modifiers.control_key();
1323        let shift = state.modifiers.shift_key();
1324
1325        if let Some(result) = pick_shape(origin, dir, &mesh_refs) {
1326            if ctrl {
1327                // Toggle clicked shape in/out of selection
1328                if state.selected_ids.contains(&result.shape_id) {
1329                    state.selected_ids.remove(&result.shape_id);
1330                    viewer_tx.send(ViewerToRepl::SelectionChanged).ok();
1331                    LAST_SELECTION.store(result.shape_id, Ordering::SeqCst);
1332                    LAST_SELECTION_ACTION.store(2, Ordering::SeqCst);
1333                } else {
1334                    state.selected_ids.insert(result.shape_id);
1335                    viewer_tx.send(ViewerToRepl::SelectionChanged).ok();
1336                    LAST_SELECTION.store(result.shape_id, Ordering::SeqCst);
1337                    LAST_SELECTION_ACTION.store(1, Ordering::SeqCst);
1338                }
1339            } else if shift {
1340                // Additive — insert if not already present
1341                if state.selected_ids.insert(result.shape_id) {
1342                    viewer_tx.send(ViewerToRepl::SelectionChanged).ok();
1343                    LAST_SELECTION.store(result.shape_id, Ordering::SeqCst);
1344                    LAST_SELECTION_ACTION.store(1, Ordering::SeqCst);
1345                }
1346            } else {
1347                // Plain click — replace selection
1348                state.selected_ids.clear();
1349                state.selected_ids.insert(result.shape_id);
1350                viewer_tx.send(ViewerToRepl::SelectionChanged).ok();
1351                LAST_SELECTION.store(result.shape_id, Ordering::SeqCst);
1352                LAST_SELECTION_ACTION.store(1, Ordering::SeqCst);
1353            }
1354        } else {
1355            // Miss — no shape under cursor
1356            if !ctrl && !shift {
1357                // Plain click on empty space: clear selection
1358                let was_selected = !state.selected_ids.is_empty();
1359                state.selected_ids.clear();
1360                if was_selected {
1361                    viewer_tx.send(ViewerToRepl::SelectionChanged).ok();
1362                    LAST_SELECTION.store(u64::MAX, Ordering::SeqCst);
1363                    LAST_SELECTION_ACTION.store(3, Ordering::SeqCst);
1364                }
1365            }
1366            // Ctrl/Shift click on empty space: no-op
1367        }
1368
1369        // Sync selection state to the global for Janet queries
1370        if let Some(selected) = SELECTED_IDS.get() {
1371            *selected.write().unwrap() = state.selected_ids.clone();
1372        }
1373    }

1375    fn check_repl_commands(rx: &Receiver<ReplToViewer>, state: &mut ViewerState) {
1376        while let Ok(cmd) = rx.try_recv() {
1377            match cmd {
1378                ReplToViewer::FitToBounds {
1379                    center,
1380                    radius,
1381                    keep_angle,
1382                } => {
1383                    Self::fit_to_bounds(state, center, radius, keep_angle);
1384                }
1385                ReplToViewer::SetViewAngles {
1386                    yaw,
1387                    pitch,
1388                    distance,
1389                } => {
1390                    let target_radius = distance.unwrap_or(state.camera.radius);
1391                    state.fit_animation = Some(FitAnimation::new(
1392                        state.camera.target,
1393                        state.camera.target,
1394                        state.camera.radius,
1395                        target_radius,
1396                        state.camera.yaw,
1397                        yaw,
1398                        state.camera.pitch,
1399                        pitch,
1400                    ));
1401                }
1402                ReplToViewer::SetWindowSize { width, height } => {
1403                    let logical = LogicalSize::new(width as f64, height as f64);
1404                    let _ = state.window.request_inner_size(logical);
1405                }
1406                ReplToViewer::SetFullscreen(fs) => {
1407                    if fs && let Some(monitor) = state.window.current_monitor() {
1408                        state
1409                            .window
1410                            .set_fullscreen(Some(Fullscreen::Borderless(Some(monitor))));
1411                    } else if !fs {
1412                        state.window.set_fullscreen(None);
1413                    }
1414                    WINDOW_FULLSCREEN.store(fs, Ordering::SeqCst);
1415                }
1416                ReplToViewer::SetMaximized(mx) => {
1417                    state.window.set_maximized(mx);
1418                    WINDOW_MAXIMIZED.store(mx, Ordering::SeqCst);
1419                }
1420                ReplToViewer::HighlightShape { id } => {
1421                    state.highlighted_shape = Some(id);
1422                    REGISTRY_GENERATION.fetch_add(1, Ordering::SeqCst);
1423                }
1424                ReplToViewer::ClearHighlight => {
1425                    state.highlighted_shape = None;
1426                    REGISTRY_GENERATION.fetch_add(1, Ordering::SeqCst);
1427                }
1428            }
1429        }
1430    }

1432    fn fit_to_bounds(state: &mut ViewerState, center: DVec3, radius: f64, keep_angle: bool) {
1433        let aspect = state.size.width as f64 / state.size.height.max(1) as f64;
1434        let margin = 1.3;
1435
1436        let target_radius = if state.camera.perspective {
1437            let tan_half_fov = (state.camera.fov_y * 0.5).tan();
1438            let d_height = radius / tan_half_fov;
1439            let d_width = radius / (tan_half_fov * aspect);
1440            d_height.max(d_width) * margin
1441        } else {
1442            let from_height = 2.0 * radius;
1443            let from_width = 2.0 * radius / aspect;
1444            from_height.max(from_width) * margin
1445        };
1446
1447        let end_yaw = if keep_angle { state.camera.yaw } else { 0.0 };
1448        let end_pitch = if keep_angle { state.camera.pitch } else { 0.4 };
1449
1450        state.fit_animation = Some(FitAnimation::new(
1451            state.camera.target,
1452            center,
1453            state.camera.radius,
1454            target_radius,
1455            state.camera.yaw,
1456            end_yaw,
1457            state.camera.pitch,
1458            end_pitch,
1459        ));
1460    }

1462    fn render(state: &mut ViewerState) {
1463        // Sync projection mode from atomic (controlled by Janet or keyboard)
1464        let target_perspective = PROJECTION_PERSPECTIVE.load(Ordering::Relaxed);
1465        if target_perspective != state.camera.perspective {
1466            state.camera.toggle_projection();
1467        }
1468
1469        // Update camera animations
1470        let now = std::time::Instant::now();
1471        let dt = (now - state.last_time).as_secs_f64();
1472        state.last_time = now;
1473        state.animation.update(&mut state.camera, dt);
1474        if let Some(ref mut fit) = state.fit_animation
1475            && fit.update(&mut state.camera, dt)
1476        {
1477            state.fit_animation = None;
1478        }
1479
1480        // Update camera uniforms
1481        let aspect = state.size.width as f64 / state.size.height.max(1) as f64;
1482        let view_proj = state.camera.matrix(aspect);
1483        let uniforms = ViewUniforms {
1484            view_proj: view_proj.to_cols_array_2d().map(|r| r.map(|v| v as f32)),
1485        };
1486        state
1487            .surface_drawer
1488            .update_uniforms(&state.queue, &uniforms);
1489        state.edge_drawer.update_uniforms(&state.queue, &uniforms);
1490        state
1491            .gizmo_renderer
1492            .update_uniforms(&state.queue, &state.device, &state.camera);
1493
1494        // Rebuild GPU data only if registry has changed (dirty tracking)
1495        let current_gen = REGISTRY_GENERATION.load(std::sync::atomic::Ordering::Relaxed);
1496        if current_gen != state.last_generation {
1497            state.last_generation = current_gen;
1498
1499            let registry = global_shape_registry();
1500            let visible = registry.visible_shapes();
1501
1502            // Rebuild surface meshes
1503            let color_layout = state.surface_drawer.color_bind_group_layout();
1504            let meshes: Vec<CadMesh> = visible
1505                .iter()
1506                .filter_map(|entry| {
1507                    entry.mesh.as_ref().map(|m| {
1508                        CadMesh::new(&state.device, m, entry.shape_id, entry.color, color_layout)
1509                    })
1510                })
1511                .collect();
1512            state.surface_drawer.set_meshes(meshes);
1513
1514            // Build SegmentInstance arrays for instanced line rendering
1515            let selected_ids = &state.selected_ids;
1516            let highlighted_id = state.highlighted_shape;
1517            let mut inactive_instances: Vec<SegmentInstance> = Vec::new();
1518            let mut active_instances: Vec<SegmentInstance> = Vec::new();
1519
1520            for entry in &visible {
1521                let is_active = selected_ids.contains(&entry.shape_id)
1522                    || highlighted_id.is_some_and(|hid| hid == entry.shape_id);
1523                let target = if is_active {
1524                    &mut active_instances
1525                } else {
1526                    &mut inactive_instances
1527                };
1528                for polyline in &entry.edge_polylines {
1529                    for pair in polyline.windows(2) {
1530                        target.push(SegmentInstance {
1531                            a: [pair[0][0] as f32, pair[0][1] as f32, pair[0][2] as f32],
1532                            b: [pair[1][0] as f32, pair[1][1] as f32, pair[1][2] as f32],
1533                            cum_length: 0.0,
1534                        });
1535                    }
1536                }
1537            }
1538
1539            // Build inactive instance buffer
1540            if !inactive_instances.is_empty() {
1541                state.inactive_instance_buffer =
1542                    state
1543                        .device
1544                        .create_buffer_init(&wgpu::util::BufferInitDescriptor {
1545                            label: Some("inactive_edge_instances"),
1546                            contents: bytemuck::cast_slice(&inactive_instances),
1547                            usage: wgpu::BufferUsages::VERTEX,
1548                        });
1549            }
1550            state.inactive_num_instances = inactive_instances.len() as u32;
1551
1552            // Build active instance buffer
1553            if !active_instances.is_empty() {
1554                state.active_instance_buffer =
1555                    state
1556                        .device
1557                        .create_buffer_init(&wgpu::util::BufferInitDescriptor {
1558                            label: Some("active_edge_instances"),
1559                            contents: bytemuck::cast_slice(&active_instances),
1560                            usage: wgpu::BufferUsages::VERTEX,
1561                        });
1562            }
1563            state.active_num_instances = active_instances.len() as u32;
1564        }
1565
1566        let frame = match state.surface.get_current_texture() {
1567            Ok(f) => f,
1568            Err(_) => return,
1569        };
1570        let view = frame
1571            .texture
1572            .create_view(&wgpu::TextureViewDescriptor::default());
1573
1574        let mut encoder = state
1575            .device
1576            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
1577                label: Some("render encoder"),
1578            });
1579
1580        // Main scene render pass
1581        {
1582            let mut pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
1583                label: Some("main pass"),
1584                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
1585                    view: &view,
1586                    resolve_target: None,
1587                    ops: wgpu::Operations {
1588                        load: wgpu::LoadOp::Clear(wgpu::Color {
1589                            r: 0.15,
1590                            g: 0.15,
1591                            b: 0.15,
1592                            a: 1.0,
1593                        }),
1594                        store: wgpu::StoreOp::Store,
1595                    },
1596                })],
1597                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
1598                    view: &state.depth_texture_view,
1599                    depth_ops: Some(wgpu::Operations {
1600                        load: wgpu::LoadOp::Clear(1.0),
1601                        store: wgpu::StoreOp::Store,
1602                    }),
1603                    stencil_ops: None,
1604                }),
1605                timestamp_writes: None,
1606                occlusion_query_set: None,
1607            });
1608
1609            // Grid (no depth test, overlay style)
1610            pass.set_bind_group(0, &state.edge_drawer.uniform_bind_group, &[]);
1611            pass.set_pipeline(&state.edge_drawer.grid_pipeline);
1612            state.grid_renderer.render(&mut pass);
1613
1614            // Mesh surfaces (depth test: Less, writes depth)
1615            state
1616                .surface_drawer
1617                .render(&mut pass, &state.selected_ids, state.highlighted_shape);
1618
1619            // Shape edges (depth test: Less with negative bias toward camera,
1620            // rendered AFTER meshes so edges overlay mesh surfaces)
1621            state.edge_drawer.render(
1622                &mut pass,
1623                &state.inactive_instance_buffer,
1624                state.inactive_num_instances,
1625                &state.active_instance_buffer,
1626                state.active_num_instances,
1627                SHOW_BACK_EDGES.load(Ordering::Relaxed),
1628            );
1629        }
1630
1631        // Gizmo overlay pass (no depth, alpha blending, top-right viewport)
1632        {
1633            let gs = state.gizmo_viewport_size;
1634            let gm = state.gizmo_margin;
1635            let gx = state.size.width.saturating_sub(gs + gm);
1636            let gy = gm;
1637
1638            let mut pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
1639                label: Some("gizmo pass"),
1640                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
1641                    view: &view,
1642                    resolve_target: None,
1643                    ops: wgpu::Operations {
1644                        load: wgpu::LoadOp::Load,
1645                        store: wgpu::StoreOp::Store,
1646                    },
1647                })],
1648                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
1649                    view: &state.gizmo_depth_view,
1650                    depth_ops: Some(wgpu::Operations {
1651                        load: wgpu::LoadOp::Clear(1.0),
1652                        store: wgpu::StoreOp::Discard,
1653                    }),
1654                    stencil_ops: None,
1655                }),
1656                timestamp_writes: None,
1657                occlusion_query_set: None,
1658            });
1659
1660            pass.set_viewport(gx as f32, gy as f32, gs as f32, gs as f32, 0.0, 1.0);
1661            pass.set_scissor_rect(gx, gy, gs, gs);
1662            state.gizmo_renderer.render(&mut pass);
1663        }
1664
1665        // Egui overlay pass
1666        {
1667            let raw_input = state.egui_state.take_egui_input(&state.window);
1668            let full_output = state.egui_ctx.run(raw_input, |ctx| {
1669                state.stats.ui(ctx, &state.camera, &state.selected_ids, dt);
1670                state.help.ui(ctx);
1671            });
1672
1673            let pixels_per_point = state.window.scale_factor() as f32;
1674            let paint_jobs = state
1675                .egui_ctx
1676                .tessellate(full_output.shapes, pixels_per_point);
1677
1678            for (id, delta) in &full_output.textures_delta.set {
1679                state
1680                    .egui_renderer
1681                    .update_texture(&state.device, &state.queue, *id, delta);
1682            }
1683            for id in &full_output.textures_delta.free {
1684                state.egui_renderer.free_texture(id);
1685            }
1686
1687            let screen_descriptor = egui_wgpu::ScreenDescriptor {
1688                size_in_pixels: [state.size.width, state.size.height],
1689                pixels_per_point,
1690            };
1691
1692            let _ = state.egui_renderer.update_buffers(
1693                &state.device,
1694                &state.queue,
1695                &mut encoder,
1696                &paint_jobs,
1697                &screen_descriptor,
1698            );
1699
1700            let mut pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
1701                label: Some("egui pass"),
1702                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
1703                    view: &view,
1704                    resolve_target: None,
1705                    ops: wgpu::Operations {
1706                        load: wgpu::LoadOp::Load,
1707                        store: wgpu::StoreOp::Store,
1708                    },
1709                })],
1710                depth_stencil_attachment: None,
1711                timestamp_writes: None,
1712                occlusion_query_set: None,
1713            });
1714
1715            // egui-wgpu requires RenderPass<'static> but our pass borrows view
1716            // which is a local. The pass is consumed immediately before view drops,
1717            // so this is safe.
1718            #[allow(deprecated)]
1719            let pass_ref = unsafe {
1720                std::mem::transmute::<&mut wgpu::RenderPass<'_>, &mut wgpu::RenderPass<'static>>(
1721                    &mut pass,
1722                )
1723            };
1724            state
1725                .egui_renderer
1726                .render(pass_ref, &paint_jobs, &screen_descriptor);
1727        }
1728
1729        state.queue.submit(Some(encoder.finish()));
1730        frame.present();
1731    }
