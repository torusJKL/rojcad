5/// Result of a ray-shape intersection test.
6#[derive(Debug, Clone)]
7pub struct PickResult {
8    pub shape_id: u64,
9    pub distance: f64,
10}
11
12/// Möller–Trumbore ray-triangle intersection algorithm.
13/// Returns `Some(t)` where `t` is the distance along the ray to the intersection point,
14/// or `None` if no intersection.
15pub fn ray_triangle_intersect(
16    origin: DVec3,
17    dir: DVec3,
18    v0: DVec3,
19    v1: DVec3,
20    v2: DVec3,
21) -> Option<f64> {
22    let edge1 = v1 - v0;
23    let edge2 = v2 - v0;
24    let h = dir.cross(edge2);
25    let a = edge1.dot(h);
26    if a.abs() < 1e-12 {
27        return None;
28    }
29    let f = 1.0 / a;
30    let s = origin - v0;
31    let u = f * s.dot(h);
32    if !(0.0..=1.0).contains(&u) {
33        return None;
34    }
35    let q = s.cross(edge1);
36    let v = f * dir.dot(q);
37    if v < 0.0 || u + v > 1.0 {
38        return None;
39    }
40    let t = f * edge2.dot(q);
41    if t > 1e-12 { Some(t) } else { None }
42}

44/// Cast a ray against all meshes and return the closest intersection.
45pub fn pick_shape(origin: DVec3, dir: DVec3, meshes: &[(u64, &MeshData)]) -> Option<PickResult> {
46    let mut closest: Option<PickResult> = None;
47    for (shape_id, mesh) in meshes {
48        for chunk in mesh.indices.chunks(3) {
49            if chunk.len() < 3 {
50                continue;
51            }
52            let i0 = chunk[0] as usize;
53            let i1 = chunk[1] as usize;
54            let i2 = chunk[2] as usize;
55            if i0 >= mesh.vertices.len() || i1 >= mesh.vertices.len() || i2 >= mesh.vertices.len() {
56                continue;
57            }
58            let v0: DVec3 = mesh.vertices[i0].map(|x| x as f64).into();
59            let v1: DVec3 = mesh.vertices[i1].map(|x| x as f64).into();
60            let v2: DVec3 = mesh.vertices[i2].map(|x| x as f64).into();
61            if let Some(t) = ray_triangle_intersect(origin, dir, v0, v1, v2) {
62                match closest.as_ref() {
63                    Some(best) if t < best.distance => {
64                        closest = Some(PickResult {
65                            shape_id: *shape_id,
66                            distance: t,
67                        });
68                    }
69                    None => {
70                        closest = Some(PickResult {
71                            shape_id: *shape_id,
72                            distance: t,
73                        });
74                    }
75                    _ => {}
76                }
77            }
78        }
79    }
80    closest
