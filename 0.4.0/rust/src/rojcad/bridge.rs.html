1//! Rust `extern "C"` bridge layer — Janet C API function declarations
2//! and Rust callbacks invoked from bridge/bridge.c.
3//!
4//! Most items here are only referenced by the C linker, not called directly
5//! from Rust code, so we suppress false-positive unused warnings.
6
7#![allow(dead_code, non_camel_case_types, non_snake_case)]
8
9use std::ffi::{c_char, c_double, c_int, c_void};

321// ── Abstract type descriptor ──────────────────────────────────────────────────
322
323#[repr(C)]
324pub struct JanetAbstractType {
325    pub name: *const c_char,
326    pub gc: Option<unsafe extern "C" fn(*mut c_void, c_size_t) -> c_int>,
327    pub gcmark: Option<unsafe extern "C" fn(*mut c_void, c_size_t) -> c_int>,
328    pub get: Option<unsafe extern "C" fn(*mut c_void, Janet, *mut Janet) -> c_int>,
329    pub put: Option<unsafe extern "C" fn(*mut c_void, Janet, Janet)>,
330    pub marshal: Option<unsafe extern "C" fn(*mut c_void, *mut JanetMarshalContext)>,
331    pub unmarshal: Option<unsafe extern "C" fn(*mut JanetMarshalContext) -> *mut c_void>,
332    pub tostring: Option<unsafe extern "C" fn(*mut c_void, *mut JanetBuffer)>,
333    pub compare: Option<unsafe extern "C" fn(*mut c_void, *mut c_void) -> c_int>,
334    pub hash: Option<unsafe extern "C" fn(*mut c_void, c_size_t) -> i32>,
335    pub next: Option<unsafe extern "C" fn(*mut c_void, Janet) -> Janet>,
336    pub call: Option<unsafe extern "C" fn(*mut c_void, i32, *mut Janet) -> Janet>,
337    pub length: Option<unsafe extern "C" fn(*mut c_void, c_size_t) -> usize>,
338    pub bytes: Option<unsafe extern "C" fn(*mut c_void, c_size_t) -> JanetByteView>,
339    pub gcperthread: Option<unsafe extern "C" fn(*mut c_void, c_size_t) -> c_int>,
340}
