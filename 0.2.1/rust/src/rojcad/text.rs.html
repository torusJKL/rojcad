12// ── Segment types from a glyph outline ────────────────────────────────────

14#[derive(Clone, Debug)]
15pub enum Segment {
16    Line((f64, f64), (f64, f64)),
17    Quad((f64, f64), (f64, f64), (f64, f64)),
18    Cubic((f64, f64), (f64, f64), (f64, f64), (f64, f64)),
19}
20
21impl Segment {
22    fn control_points(&self) -> Vec<(f64, f64)> {
23        match self {
24            Segment::Line(p0, p1) => vec![*p0, *p1],
25            Segment::Quad(p0, p1, p2) => vec![*p0, *p1, *p2],
26            Segment::Cubic(p0, p1, p2, p3) => vec![*p0, *p1, *p2, *p3],
27        }
28    }
29}
30
31#[derive(Clone, Debug)]
32pub struct Contour {
33    pub segments: Vec<Segment>,
34}
35
36impl Contour {
37    fn bounds(&self) -> Option<(f64, f64, f64, f64)> {
38        let mut pts = self.segments.iter().flat_map(|s| s.control_points()).fuse();
39        let (x0, y0) = pts.next()?;
40        let (mut x_min, mut x_max, mut y_min, mut y_max) = (x0, x0, y0, y0);
41        for (x, y) in pts {
42            if x < x_min {
43                x_min = x;
44            }
45            if x > x_max {
46                x_max = x;
47            }
48            if y < y_min {
49                y_min = y;
50            }
51            if y > y_max {
52                y_max = y;
53            }
54        }
55        Some((x_min, y_min, x_max, y_max))
56    }
57
58    fn area(&self) -> f64 {
59        self.bounds()
60            .map(|(x1, y1, x2, y2)| (x2 - x1) * (y2 - y1))
61            .unwrap_or(0.0)
62    }
63}

67#[derive(Debug)]
68pub struct FontData {
69    data: Vec<u8>,
70    pub units_per_em: u16,
71    pub ascender: i16,
72    pub descender: i16,
73}
74
75impl FontData {
76    pub fn from_bytes(data: Vec<u8>) -> Result<Self, String> {
77        let face = ttf_parser::Face::parse(&data, 0).map_err(|e| format!("invalid font: {e:?}"))?;
78        Ok(Self {
79            units_per_em: face.units_per_em(),
80            ascender: face.ascender(),
81            descender: face.descender(),
82            data,
83        })
84    }
85
86    pub fn from_path(path: &str) -> Result<Self, String> {
87        // 1. Try direct path first
88        if let Ok(data) = std::fs::read(path) {
89            return Self::from_bytes(data);
90        }
91
92        // 2. Scan system font dirs (cached) and try to resolve
93        let resolved = resolve_font_path(path);
94        match resolved {
95            Some(resolved_path) => {
96                let data = std::fs::read(&resolved_path)
97                    .map_err(|e| format!("cannot read font '{resolved_path}': {e}"))?;
98                Self::from_bytes(data)
99            }
100            None => Err(format!(
101                "cannot read font '{path}': No such file or directory"
102            )),
103        }
104    }
105
106    pub fn face(&self) -> ttf_parser::Face<'_> {
107        ttf_parser::Face::parse(&self.data, 0).unwrap()
108    }
109
110    #[expect(dead_code)]
111    pub fn font_name(&self) -> Option<String> {
112        let face = self.face();
113        // Prefer Windows+UnicodeBMP, name_id=1 (family)
114        for n in face.names() {
115            if n.name_id == 1
116                && n.platform_id == ttf_parser::PlatformId::Windows
117                && n.encoding_id == 1
118                && let Some(s) = n.to_string()
119            {
120                return Some(s);
121            }
122        }
123        // fallback: any unicode name with name_id=1
124        for n in face.names() {
125            if n.name_id == 1
126                && let Some(s) = n.to_string()
127            {
128                return Some(s);
129            }
130        }
131        None
132    }
133}

135// ── Glyph outline extraction via OutlineBuilder ────────────────────────────

137struct GlyphBuilder {
138    contours: Vec<Contour>,
139    current: Vec<Segment>,
140    first: Option<(f64, f64)>,
141    cursor: Option<(f64, f64)>,
142}
143
144impl GlyphBuilder {
145    fn new() -> Self {
146        Self {
147            contours: Vec::new(),
148            current: Vec::new(),
149            first: None,
150            cursor: None,
151        }
152    }
153
154    fn close_contour(&mut self) {
155        if !self.current.is_empty() {
156            self.contours.push(Contour {
157                segments: std::mem::take(&mut self.current),
158            });
159        }
160        self.first = None;
161        self.cursor = None;
162    }
163
164    fn push(&mut self, s: Segment) {
165        match &s {
166            Segment::Line(_, e) | Segment::Quad(_, _, e) | Segment::Cubic(_, _, _, e) => {
167                self.cursor = Some(*e);
168            }
169        }
170        self.current.push(s);
171    }
172}
173
174impl ttf_parser::OutlineBuilder for GlyphBuilder {
175    fn move_to(&mut self, x: f32, y: f32) {
176        self.close_contour();
177        let p = (x as f64, y as f64);
178        self.first = Some(p);
179        self.cursor = Some(p);
180    }
181    fn line_to(&mut self, x: f32, y: f32) {
182        if let Some(s) = self.cursor {
183            self.push(Segment::Line(s, (x as f64, y as f64)));
184        }
185    }
186    fn quad_to(&mut self, x1: f32, y1: f32, x: f32, y: f32) {
187        if let Some(s) = self.cursor {
188            self.push(Segment::Quad(
189                s,
190                (x1 as f64, y1 as f64),
191                (x as f64, y as f64),
192            ));
193        }
194    }
195    fn curve_to(&mut self, x1: f32, y1: f32, x2: f32, y2: f32, x: f32, y: f32) {
196        if let Some(s) = self.cursor {
197            self.push(Segment::Cubic(
198                s,
199                (x1 as f64, y1 as f64),
200                (x2 as f64, y2 as f64),
201                (x as f64, y as f64),
202            ));
203        }
204    }
205    fn close(&mut self) {
206        if let (Some(fp), Some(cp)) = (self.first, self.cursor)
207            && ((cp.0 - fp.0).abs() > 1e-10 || (cp.1 - fp.1).abs() > 1e-10)
208        {
209            self.push(Segment::Line(cp, fp));
210        }
211        self.close_contour();
212    }
213}

215pub fn glyph_outlines(font: &FontData, gid: ttf_parser::GlyphId) -> Option<Vec<Contour>> {
216    let face = font.face();
217    let mut builder = GlyphBuilder::new();
218    let ok = face.outline_glyph(gid, &mut builder);
219    builder.close_contour();
220    ok.map(|_| builder.contours)
221}

231pub fn layout_text(font: &FontData, text: &str, size: f64) -> Result<Vec<PositionedGlyph>, String> {
232    if text.is_empty() {
233        return Err("text: empty string".to_string());
234    }
235    let upem = font.units_per_em as f64;
236    if upem == 0.0 {
237        return Err("text: font has zero units-per-em".to_string());
238    }
239    let scale = size / upem;
240    let face = font.face();
241    let mut result = Vec::new();
242    let mut cx = 0.0;
243
244    for ch in text.chars() {
245        let gid = face.glyph_index(ch).unwrap_or(ttf_parser::GlyphId(0));
246        let advance = face.glyph_hor_advance(gid).unwrap_or(0) as f64 * scale;
247        if let Some(contours) = glyph_outlines(font, gid)
248            && !contours.is_empty()
249        {
250            result.push(PositionedGlyph {
251                contours,
252                x_offset: cx,
253            });
254        }
255        cx += advance;
256    }
257
258    if result.is_empty() {
259        return Err("text: no visible glyphs in string".to_string());
260    }
261
262    Ok(result)
263}

267fn seg_to_edges(seg: &Segment, scale: f64, x_off: f64, y_off: f64, wp: &Workplane) -> Edge {
268    let to_world =
269        |x: f64, y: f64| wp.to_world_pos(DVec3::new(x * scale + x_off, y * scale + y_off, 0.0));
270    match *seg {
271        Segment::Line(p0, p1) => Edge::segment(to_world(p0.0, p0.1), to_world(p1.0, p1.1)),
272        Segment::Quad(p0, p1, p2) => Edge::bezier([
273            to_world(p0.0, p0.1),
274            to_world(p1.0, p1.1),
275            to_world(p2.0, p2.1),
276        ]),
277        Segment::Cubic(p0, p1, p2, p3) => Edge::bezier([
278            to_world(p0.0, p0.1),
279            to_world(p1.0, p1.1),
280            to_world(p2.0, p2.1),
281            to_world(p3.0, p3.1),
282        ]),
283    }
284}

286fn contour_to_edges(
287    contour: &Contour,
288    scale: f64,
289    x_off: f64,
290    y_off: f64,
291    wp: &Workplane,
292) -> Vec<Edge> {
293    contour
294        .segments
295        .iter()
296        .map(|s| seg_to_edges(s, scale, x_off, y_off, wp))
297        .collect()
298}

300/// Split contours into outer (largest area) and holes (rest).
301fn split_outer_holes(contours: &[Contour]) -> (usize, Vec<usize>) {
302    if contours.len() <= 1 {
303        return (0, vec![]);
304    }
305    let outer_idx = contours
306        .iter()
307        .enumerate()
308        .max_by(|(_, a), (_, b)| a.area().partial_cmp(&b.area()).unwrap())
309        .map(|(i, _)| i)
310        .unwrap_or(0);
311    let holes: Vec<usize> = (0..contours.len()).filter(|i| *i != outer_idx).collect();
312    (outer_idx, holes)
313}

317/// Build a 2D text Shape (Face or Compound of Faces) from a string.
318///
319/// Each glyph with holes (e.g. 'A', 'B', '8') has holes punched via
320/// boolean subtract. Multiple glyphs are grouped into a Compound.
321pub fn text_to_shape(
322    text: &str,
323    font: &FontData,
324    size: f64,
325    wp: &Workplane,
326) -> Result<Shape, String> {
327    let positioned = layout_text(font, text, size)?;
328    let upem = font.units_per_em as f64;
329    let scale = size / upem;
330
331    // Y-offset to center text vertically: baseline at -(descender + ascender)/2
332    let asc = font.ascender as f64 * scale;
333    let desc = font.descender as f64 * scale;
334    let y_off = -(asc + desc) / 2.0 - desc;
335
336    let mut glyph_shapes: Vec<Shape> = Vec::new();
337
338    for pg in &positioned {
339        let (outer_idx, hole_idxs) = split_outer_holes(&pg.contours);
340
341        if outer_idx >= pg.contours.len() || pg.contours[outer_idx].segments.is_empty() {
342            continue;
343        }
344
345        let outer_edges = contour_to_edges(&pg.contours[outer_idx], scale, pg.x_offset, y_off, wp);
346        let outer_wire = Wire::from_edges(&outer_edges);
347        let outer_face = Face::from_wire(&outer_wire);
348        let outer_shape = Shape::from(outer_face);
349
350        let hole_shapes: Vec<Shape> = hole_idxs
351            .into_iter()
352            .filter(|i| !pg.contours[*i].segments.is_empty())
353            .map(|i| {
354                let edges = contour_to_edges(&pg.contours[i], scale, pg.x_offset, y_off, wp);
355                let wire = Wire::from_edges(&edges);
356                let face = Face::from_wire(&wire);
357                Shape::from(face)
358            })
359            .collect();
360
361        if hole_shapes.is_empty() {
362            glyph_shapes.push(outer_shape);
363            continue;
364        }
365
366        let mut current = outer_shape;
367        for hole in &hole_shapes {
368            current = current.subtract(hole).shape;
369        }
370        glyph_shapes.push(current);
371    }
372
373    if glyph_shapes.is_empty() {
374        return Err("text: no glyphs produced geometry".to_string());
375    }
376
377    // Combine into a single Shape
378    if glyph_shapes.len() == 1 {
379        Ok(glyph_shapes.into_iter().next().unwrap())
380    } else {
381        let compound = Compound::from_shapes(&glyph_shapes);
382        Ok(Shape::from(compound))
383    }
384}

386/// Build an extruded 3D text Shape from a string.
387pub fn text_to_solid(
388    text: &str,
389    font: &FontData,
390    size: f64,
391    depth: f64,
392    both: bool,
393    wp: &Workplane,
394) -> Result<Shape, String> {
395    let positioned = layout_text(font, text, size)?;
396    let upem = font.units_per_em as f64;
397    let scale = size / upem;
398
399    let asc = font.ascender as f64 * scale;
400    let desc = font.descender as f64 * scale;
401    let y_off = -(asc + desc) / 2.0 - desc;
402
403    let mut solid_shapes: Vec<Shape> = Vec::new();
404
405    for pg in &positioned {
406        let (outer_idx, hole_idxs) = split_outer_holes(&pg.contours);
407
408        if outer_idx >= pg.contours.len() || pg.contours[outer_idx].segments.is_empty() {
409            continue;
410        }
411
412        let outer_edges = contour_to_edges(&pg.contours[outer_idx], scale, pg.x_offset, y_off, wp);
413        let outer_wire = Wire::from_edges(&outer_edges);
414        let outer_face = Face::from_wire(&outer_wire);
415
416        let hole_shapes: Vec<Shape> = hole_idxs
417            .into_iter()
418            .filter(|i| !pg.contours[*i].segments.is_empty())
419            .map(|i| {
420                let edges = contour_to_edges(&pg.contours[i], scale, pg.x_offset, y_off, wp);
421                Shape::from(Face::from_wire(&Wire::from_edges(&edges)))
422            })
423            .collect();
424
425        let mut current_shape = Shape::from(&outer_face);
426        for hole in &hole_shapes {
427            current_shape = current_shape.subtract(hole).shape;
428        }
429
430        let dir = DVec3::new(0.0, 0.0, depth);
431        let half_z = DVec3::new(0.0, 0.0, depth / 2.0);
432
433        for face in current_shape.faces() {
434            let solid = if both {
435                let shifted = Shape::from(&face).translated(-half_z);
436                shifted.expect_face().extrude(dir)
437            } else {
438                face.extrude(dir)
439            };
440            solid_shapes.push(Shape::from(solid));
441        }
442    }
443
444    if solid_shapes.is_empty() {
445        return Err("text: no glyphs produced geometry".to_string());
446    }
447
448    if solid_shapes.len() == 1 {
449        Ok(solid_shapes.into_iter().next().unwrap())
450    } else {
451        Ok(Shape::from(Compound::from_shapes(&solid_shapes)))
452    }
453}

476fn scan_font_dirs() -> Vec<PathBuf> {
477    let mut dirs = Vec::new();
478
479    // Linux / BSD
480    if cfg!(target_os = "linux") || cfg!(target_os = "freebsd") || cfg!(target_os = "netbsd") {
481        dirs.push(PathBuf::from("/usr/share/fonts"));
482        dirs.push(PathBuf::from("/usr/local/share/fonts"));
483        if let Ok(ref home) = std::env::var("HOME") {
484            dirs.push(PathBuf::from(home).join(".fonts"));
485            dirs.push(PathBuf::from(home).join(".local/share/fonts"));
486        }
487    }
488
489    // macOS
490    if cfg!(target_os = "macos") {
491        dirs.push(PathBuf::from("/Library/Fonts"));
492        dirs.push(PathBuf::from("/System/Library/Fonts"));
493        if let Ok(ref home) = std::env::var("HOME") {
494            dirs.push(PathBuf::from(home).join("Library/Fonts"));
495        }
496    }
497
498    // Windows
499    if cfg!(target_os = "windows") {
500        if let Ok(root) = std::env::var("WINDIR") {
501            dirs.push(PathBuf::from(root).join("Fonts"));
502        }
503        if let Ok(local) = std::env::var("LOCALAPPDATA") {
504            dirs.push(PathBuf::from(local).join("Microsoft/Windows/Fonts"));
505        }
506    }
507
508    dirs
509}

511fn collect_font_files(dirs: &[PathBuf]) -> Vec<PathBuf> {
512    let mut files = Vec::new();
513    for dir in dirs {
514        if !dir.exists() {
515            continue;
516        }
517        if let Ok(entries) = std::fs::read_dir(dir) {
518            for entry in entries.flatten() {
519                let path = entry.path();
520                if path.is_dir() {
521                    files.append(&mut collect_font_files(&[path]));
522                } else if let Some(ext) = path.extension().and_then(|e| e.to_str())
523                    && (ext.eq_ignore_ascii_case("ttf")
524                        || ext.eq_ignore_ascii_case("otf")
525                        || ext.eq_ignore_ascii_case("ttc"))
526                {
527                    files.push(path);
528                }
529            }
530        }
531    }
532    files
533}

535pub fn list_system_fonts() -> Vec<(String, String, FontAspect)> {
536    let dirs = scan_font_dirs();
537    let files = collect_font_files(&dirs);
538    let mut seen = HashSet::new();
539    let mut results = Vec::new();
540
541    for path in &files {
542        let path_str = path.to_string_lossy().to_string();
543        // Try to parse the font and get its name
544        let data = match std::fs::read(path) {
545            Ok(d) => d,
546            Err(_) => continue,
547        };
548        let face = match ttf_parser::Face::parse(&data, 0) {
549            Ok(f) => f,
550            Err(_) => continue,
551        };
552
553        // Get font family name
554        let name = get_font_name_from_face(&face);
555
556        // Determine aspect
557        let aspect = if face.is_bold() && face.is_italic() {
558            FontAspect::BoldItalic
559        } else if face.is_bold() {
560            FontAspect::Bold
561        } else if face.is_italic() {
562            FontAspect::Italic
563        } else {
564            FontAspect::Regular
565        };
566
567        // Deduplicate by (name, aspect)
568        let key = (name.clone(), aspect.as_str().to_string());
569        if seen.insert(key) {
570            results.push((name, path_str, aspect));
571        }
572    }
573
574    results.sort_by(|a, b| a.0.cmp(&b.0));
575    results
576}

578fn get_font_name_from_face(face: &ttf_parser::Face<'_>) -> String {
579    for n in face.names() {
580        if n.name_id == 1
581            && n.platform_id == ttf_parser::PlatformId::Windows
582            && n.encoding_id == 1
583            && let Some(s) = n.to_string()
584        {
585            return s;
586        }
587    }
588    for n in face.names() {
589        if n.name_id == 1
590            && let Some(s) = n.to_string()
591        {
592            return s;
593        }
594    }
595    String::new()
596}

598/// Read the family name from a font file, lowercased.
599fn read_font_name(path: &Path) -> Option<String> {
600    let data = std::fs::read(path).ok()?;
601    let face = ttf_parser::Face::parse(&data, 0).ok()?;
602    let name = get_font_name_from_face(&face);
603    if name.is_empty() {
604        None
605    } else {
606        Some(name.to_lowercase())
607    }
608}

610/// Cached font lookup: (filename→path, name→path).
611fn font_lookup() -> &'static Vec<(String, String)> {
612    static LOOKUP: OnceLock<Vec<(String, String)>> = OnceLock::new();
613    LOOKUP.get_or_init(|| {
614        let files = collect_font_files(&scan_font_dirs());
615        let mut entries = Vec::new();
616
617        for path in &files {
618            let path_str = path.to_string_lossy().to_string();
619            // Index by filename, e.g. "DejaVuSans.ttf"
620            if let Some(name) = path.file_name().and_then(|n| n.to_str()) {
621                entries.push((name.to_lowercase(), path_str.clone()));
622            }
623            // Index by font family name, e.g. "DejaVu Sans"
624            if let Some(name) = read_font_name(path) {
625                entries.push((name, path_str));
626            }
627        }
628
629        entries
630    })
631}

633/// Resolve a font path by trying: direct, filename lookup, font name lookup.
634fn resolve_font_path(input: &str) -> Option<String> {
635    let lower = input.to_lowercase();
636    for (key, path) in font_lookup() {
637        if *key == lower {
638            return Some(path.clone());
639        }
640    }
641    None
642}

699    fn test_hole_detection_outer_largest() {
700        let data = std::fs::read("/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf");
701        if let Some(font_data) = data.ok().and_then(|d| FontData::from_bytes(d).ok())
702            && let Some(gid) = font_data.face().glyph_index('A')
703            && let Some(contours) = glyph_outlines(&font_data, gid)
704        {
705            let (outer_idx, hole_idxs) = split_outer_holes(&contours);
706            assert!(outer_idx < contours.len());
707            assert!(!hole_idxs.is_empty(), "'A' should have hole contours");
708            let outer_area = contours[outer_idx].area();
709            for &hi in &hole_idxs {
710                assert!(
711                    outer_area > contours[hi].area(),
712                    "outer {outer_area} should be > hole area {}",
713                    contours[hi].area()
714                );
715            }
716        }
717    }
