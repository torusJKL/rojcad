pub enum ReplToViewer {
    FitToBounds {
        center: DVec3,
        radius: f64,
        keep_angle: bool,
    },
    SetViewAngles {
        yaw: f64,
        pitch: f64,
        distance: Option<f64>,
    },
    SetWindowSize {
        width: u32,
        height: u32,
    },
    SetFullscreen(bool),
    SetMaximized(bool),
}
Expand description

Commands sent from the REPL thread to the viewer thread. The viewer polls these each frame via an mpsc receiver.
